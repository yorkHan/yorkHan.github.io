    Vue.component(‘editable-span’,{
    props:[‘value’], //声明从外部接收一个value，外面给的属性
    template://模板 <div> <span v-show="!editing"></span> <input type="text" v-show="editing" v-bind:value="value" v-on:input="triggerEdit"> //调用函数x，不能用v-model="value"，不能改外边的值 <button v-on:click="$emit('edit')">编辑</button> //在template里面$emit不需要加this </div>,
    data(){ //自己的属性
    return {
    editing:false //变量表示状态
    }
    }，
    methods：{
    triggerEdit(e){
    this.$emit(‘edit’,e.target.value) //触发edit事件,传input.value当前值，在methods里面$emit需要加this
    }
    }
    })
